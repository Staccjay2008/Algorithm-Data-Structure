
'''python
class Solution:
    def fib(self, N: int) -> int:
        lst = [None] * (N+1)
        if N == 0:
            return 0
        if N == 1:
            return 1
        else:
            for i in range(N+1):
                if i == 0:
                    lst[i] = 0
                elif i == 1:
                    lst[i] = 1
                else:
                    lst[i] = lst[i-1] + lst[i-2]
        return lst[N]
